 51 	loadJSON : function(jsonURL, storage, fn){ 
 52 		var _this = this;
 53     $.ajax({
 54         type: "get",
 55         beforeSend: function(xhr){
 56     				if (xhr.overrideMimeType){
 57 				      xhr.overrideMimeType("application/json");
 58     				}
 59   			},
 60         url: jsonURL,
 61         dataType: 'json',
 62         success: function(data){  
 63             //alert("got " + jsonURL);
 64             _this[storage] = data;  
 65             
 66             //alert(JSON.stringify(_this.json_data))
 67             _this._d++; 
 68             if (_this._d === Object.size(_this.options.files)){ 
 69             	console.log('done'); 
 70             	// call
 71             	if(_this.call_back !== undefined){
 72             		_this.call_back[fn]();
 73             	}else{
 74             		fn();
 75             	}
 76             	
 77             }
 78         },
 79         error: function(e){
 80         	window.location = "/login"; 
 81 					var err = new Error('Could not catch data');
 82 				}
 83     });
 84 	},

120 	/**
121 	* Returns a given number of video stream ids that have been randomly selected
122 	*/
123 	getRandomStreams : function(exception, limit){ 
124 		var 
125 			array = [],
126 			result = {};
127 		// generate array with all stream ids  
128 		$.each(this.json_data.stream, function(i, val){
129 			if (val.id != exception){  
130 				array.push(val.id); 
131 			}
132 		});
133 		// now shuffle the array by using the Fisher-Yates shuffle algorithm
134 		for (var i = array.length - 1; i > 0; i--) {
135 			var j = Math.floor(Math.random() * (i + 1));
136 			var temp = array[i];
137 			array[i] = array[j];
138 			array[j] = temp;
139 		}
140 		// reduce the number of elements regarding the given limit
141 		array = array.slice(0, limit);
142 		// resort keys and values
143 		for(var k = 0; k < array.length; k++){
144 			result[array[k]] = 1; 
145 		}
146 		return result;
147 	},
