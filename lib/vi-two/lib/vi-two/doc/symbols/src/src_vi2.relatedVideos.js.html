 52 		determineCriteria : function(id){
 53 			var _this = this;
 54 			var streams = []; 
 55 			$.each( this.options.criteria, function(i, criterion){ 
 56 				switch(criterion.criterion){
 57 					case "random-destructor":
 58 						_this.weightResults( vi2.db.getRandomStreams(id, _this.options.limit), criterion.weight );	
 59 						break;
 60 					case "incomming-links" :
 61 						_this.weightResults( vi2.db.getLinkSourcesById(id), criterion.weight );	
 62 						break;
 63 					case "outgoing-links" :
 64 						_this.weightResults( vi2.db.getLinkTargetsById(id), criterion.weight );	
 65 						break;
 66 					case "same-author"	:
 67 						_this.weightResults( vi2.db.getStreamsOfSameAuthor(id), criterion.weight );	
 68 						break;
 69 					case "same-tags"	: 
 70 						_this.weightResults( vi2.db.getStreamsWithSameTag(id), criterion.weight);	
 71 						break;
 72 					default :
 73 						// do nothing			
 74 				}
 75 			}); 
 76 			// render results
 77 			this.showRelatedVideos();
 78 		},

 98 		sortByRelevance : function(arr){
 99 			var sortable = [];
100 			for (var el in arr){

102 			}			
103 			return sortable.sort(function(a, b) { 
104 				return  b[1] - a[1]; 
105 			});
