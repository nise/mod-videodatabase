 20 	var Log = $.inherit(/** @lends Log# */{
 21 
 22 		/** constructs */
 23   	__constructor : function(options) { 
 24   			var _this = this;
 25   			this.options = $.extend(this.options, options); 
 26   			// get client IP
 27   			$.ajax({
 28   				url: 'php/ip.php',
 29    				success: function(res){ 
 30 						_this.ip = res.ip;
 31 					},
 32   				dataType: 'json'
 33 				});			
 34 				// clear
 35 				$('#debug').html('');
 36 		},
 37 		
 38 		name : 'log',
 39 		options : {output: 'logfile', debug_selector: '#debug', logfile:'log.txt', parameter: 'time,ip,msg,user'}, // output: debug/logfile
 40 		bucket : '',
 41 		ip : '',
 42 	
 43 		/* ... */
 44 		init : function(){},		
 45 		
 46 		/* -- */
 47 		add : function(msg){
 48 			var logEntry = this.getLogTime()+', '+this.getIP()+', '+msg+', '+this.getUser()+'\n';
 49 			
 50 			// handle output
 51 			switch(this.options.output){
 52 				case 'debug' :
 53 					$(this.options.debug_selector).append(logEntry);
 54 					break;
 55 				case 'logfile' :
 56 					this.writeLog(logEntry);
 57 					break;
 58 			}
 59 			
 60 			// fill bucket for internal usage	
 61 			this.bucket += logEntry;
 62 		},
 63 		
 64 		/* -- */
 65 		getLogs : function(){
 66 			return this.bucket;
 67 		},
 68 		
 69 		/* -- */
 70 		getLogTime : function(){
 71 			var date = new Date();
 72 			var s = date.getSeconds();
 73 			var mi =date.getMinutes();
 74 			var h = date.getHours();
 75 			var d = date.getDate();
 76     	var m = date.getMonth()+1;
 77     	var y = date.getFullYear();
 78     	return date.getTime()+', ' + y +'-'+ (m<=9?'0'+m:m) +'-'+ (d<=9?'0'+d:d)+', '+(h<=9?'0'+h:h)+':'+(mi<=9?'0'+mi:mi)+':'+(s<=9?'0'+s:s)+':'+date.getMilliseconds();
 79 			//return date.getTime();
 80 		},
 81 		
 82 		/* -- */
 83 		getIP : function(){
 84 			return this.ip;
 85 		},
 86 						
 87 		/* -- */
 88 		getUser : function(){
 89 		 var ua = $.browser; //return '';
 90   		return	navigator.userAgent;
 91 		},
 92 		
 93 		/* -- */
 94 		writeLog : function (entry){
 95 			$.post('php/log.php', { entry:entry }); 
 96 		}					
 97 				
 98 	
 99 		
100 	}); // end class Log
