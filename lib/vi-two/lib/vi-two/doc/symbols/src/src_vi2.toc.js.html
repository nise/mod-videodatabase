 16 Vi2.TableOfContents = $.inherit(Annotation, /** @lends Vi2.TableOfContents# */{ // 
 17 
 18 		/** @constructs
 19 		*		@extends Annotation
 20 		*		@param {object} options An object containing the parameters
 21 		*		@param {boolean} options.hasTimelineMarker Whether the TOC should be annotated on the timeline or not.
 22 		*		@param {boolean} options.hasMenu Wether the TOC should be listed in a menu or not.
 23 		*		@param {string} options.menuSelector Class or id of the DOM element for the menu.
 24 		*		@param {sring} options.timelineSelector Class or id of the DOM element for the annotated timeline.
 25 		*		@param {string} options.path Path to folder where user icons are stored.
 26 		*/
 27   	__constructor : function(options) { 
 28   			this.options = $.extend(this.options, options);  
 29 		},
 30 		
 31 		name : 'toc',
 32 		type : 'annotation',
 33 		options : {
 34 			hasTimelineMarker: true, 
 35 			hasMenu : true,
 36 			menuSelector: '#toc', 
 37 			timelineSelector : '.vi2-video-seek',
 38 			path:'/'
 39 		},
 40 		currentTocElement : 0,
 41 		elements : [],
 42 
 43 		/**
 44 		Initializes the table of content and handles options
 45 		*/
 46 		init : function(annotations){
 47 			var _this = this;
 48 			// prepare toc data
 49 			var e = {}; e.tags = []; e.tags.occ = [];
 50 			$.each(annotations, function(i, val){
 51 				if(val.type == 'toc'){ 
 52 					if( val.note !== "missing"){
 53 						_this.elements.push(val.t1);
 54 					}	
 55 					e.tags.push({name: val.title, occ:[val.t1]});
 56 				}
 57 			});
 58 			// 
 59 			if( this.options.hasMenu ){
 60 				this.buildMenu(e.tags)
 61 			}	
 62 			// 
 63 			if( this.options.hasTimelineMarker ){
 64 				this.buildTimelineMarkers(e);
 65 			}
 66 					
 67 			// update toc highlight on time update
 68 			vi2.observer.player.video.addEventListener('timeupdate', function(e) { 
 69 				// reset highlight
 70 				//		$(_this.options.menuSelector+' li').each(function(i, val){ $(this).removeClass('toc-hightlight');})
 71 				// highlight toc entry
 72 				//		$(_this.options.menuSelector+ ' li#t'+this.formatTime(obj.content.target)).addClass('toc-highlight');
 73 			});
 74 		},		
 75 		
 76 		
 77 		/**
 78 		Loads toc data from database in order to generate corresponding DOM elements.
 79 		@ Resulting format of DOM element: <div type="toc" starttime=83 duration=1 id="">Objectives of the lecture</div>
 80 		*/
 81 		appendToDOM : function(id){ 
 82 			var _this = this; 
 83 			$(vi2.dom).find('[type="toc"]').each(function(i,val){ $(this).remove(); });
 84 			$.each(	vi2.db.getTocById(id), function( i, val ){  
 85 				var toc = $('<div></div>')
 86 					.attr('type',"toc")
 87 					.attr('note', val.note )
 88 					.attr('starttime', val.start)
 89 					.attr('duration', 10)
 90 					.attr('author', 'none')
 91 					.attr('id', "")
 92 					.text( decodeURIComponent(val.label) )
 93 					.appendTo( vi2.dom )
 94 					;
 95 			});
 96 		},					
 97 				
 98 		/** 
 99 		During the playback corresponding toc entries will be highlighted.
100 		*/
101 		begin : function(e, id, obj){ 
102 				// reset highlight
103 				$(this.options.menuSelector+' li').each(function(i, val){ $(this).removeClass('toc-highlight') ;})
104 				// highlight toc entry
105 				$('.toc-'+obj.displayPosition.t1).addClass('toc-highlight');
106 			
107 		},
108 	
109 		/** 
110 		Terminates time-depended toc entries
111 		*/
112 		end : function(e, id, obj){ 
113 			//$('.toc-'+obj.displayPosition.t1).removeClass('toc-highlight');
114 		},
115 		
116 		
117 		/**
118 		Builds a list menu of all entries of the table of content
119 		*/
120 		buildMenu : function(tocData){
121 			var _this = this;
122 			var toc = $('<ul></ul>').addClass('toc-list'); 
123 			
124 			$.each(tocData, function(i, val){
125 					var a = $('<a></a>')
126 					.text( val.name )
127 					.addClass('id-'+ val.occ[0])
128 					.attr('href', '#!t=npt:' + val.occ[0] + '') // former: main.options.id
129 					;				
130 					var li = $('<li></li>')
131 						.addClass('toc-'+val.occ[0])
132 						.attr('id', ''+ i)
133 						//.css('list-style-image',  "url('"+_this.options.path+"user-"+val.author+".png')")
134 						.html(a)
135 						.appendTo( toc )
136 						; 
137 					if( val.note === "missing"){
138 						li.addClass('toc-disabled');
139 					}else{
140 						li.click(function(){
141 							vi2.observer.log('clicktocfromlist:'+val.name.replace(/,/g,'##') +' '+val.author+' '+ val.occ[0]); 
142 							vi2.observer.player.currentTime( val.occ[0] );
143 							_this.currentTocElement = i;
144 						});	
145 					}	
146 			});
147 			
148 			
149 			// sort list entries by time and append them
150 			toc
151 				.find('li').tsort( { attr:"id" } )  // tsort is error prune under chromium
152 				.appendTo( this.options.menuSelector )		
153 			
154 		},
155 		
156 		
157 		/** 
158 		Displays a table of content as markers on the timeline
159 		@todo this function could to be moved to the player or separated into a timeline class
160 		*/
161 		buildTimelineMarkers : function(tocData){  
162 			var _this= this; 
163 				// display toc occurence on timeline to motivate further selection
164 				var f = function(_left, _name){ 
165 					return sp = $('<span></span>')					
166 						.addClass('toc-timeline-marker ttoc')
167 						.attr('style','left:'+_left+'px;')
168 						.tooltip({delay: 0, showURL: false, bodyHandler: function() { return $('<span></span>').text(_name);} });
169 				};
170 								
171 				$.each(tocData, function(i, val){ 
172 					var progress = this.occ[0] / vi2.observer.player.duration();
173 					progress = ((progress) * $(_this.options.timelineSelector).width());
174   	    	if (isNaN(progress) || progress > $(_this.options.timelineSelector).width()) { return;}
175 	 				$(_this.options.timelineSelector).append(f(progress, val.name));
176  				});
177 		},
178 		
179 		
180 		/** 
181 		Jumps to the next element of the table of content 
182 		*/
183 		nextElement : function(){
184 			//$('.toc-'+ (this.currentTocElement + 1) ).click();
185 			this.currentTocElement = this.currentTocElement < this.elements.length ? this.currentTocElement + 1 : this.currentTocElement;
186 			vi2.observer.player.currentTime( this.elements[ this.currentTocElement ] );
187 			//
188 		},
189 		
190 		
191 		/** 
192 		Jumps to the previous element of the table of content. 
193 		*/
194 		previousElement : function(){
195 			this.currentTocElement = this.currentTocElement == 0 ? this.currentTocElement : this.currentTocElement -1 ;
196 			vi2.observer.player.currentTime( this.elements[ this.currentTocElement ] );
197 		}
198 		
199 	}); // end class
