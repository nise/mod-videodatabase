479 	buildDropDown : function(options, label, type, func){ 
480 		var _this = this;
481 		var sel = $('<div></div>')
482 		.attr('class','nav')
483 		.text(label)
484 		.append($('<ul></ul>').addClass('nav-list').addClass(type).addClass('nav-list-'+label.toLowerCase()));
485 		
486 		$.each(options, function(i, val){
487 			var tugs = val.first_level; 
488 			$(val.second_level).each(function(i, val){ tugs = tugs+'+'+val;});	
489 			
490 			var item = $('<li></li>'); 
491 			$(sel).find('ul').append(item);
492 			item.addClass('nav-item')
493 				.html(
494 					$('<a></a>')
495 						.text(val.first_level)
496 						.click(function(){
497 							window.vi2.observer[func](type, tugs); 
498 					})
499 				);
500 						
501 			if(val.second_level == null){
502 				item.click(function(){ 
503 					window.vi2.observer[func](type, val.first_level); 				
504 				}); 
505 			}else{
506 				var nav = $('<div></div>').addClass('nav-list2').remove();
507 				$(val.second_level).each(function(i, value){ 
508 						var el = $('<a></a>')
509 							.addClass('nav-item2')
510 							.text(value)
511 							.click(function(){
512 								window.vi2.observer[func](type, value);
513 							});
514 						nav.append(el);
515 					});
516 				item.append(nav);
517 			}
518 		});		
519 		return sel;
520 	},
