253   		parseWikiVideo : function(str){
254 
255 					var arr = [];
256 					var url, start, duration, id = '_';
257   				str = str
258   					.replace(/^\[\[Video:/, '') // start delimiter
259   					.replace(/\]\]/, '') // end delimiter
260   					.replace(/\# /, ' #') // start-time
261   					.replace(/\| /, ' |') // duration
262   					.replace(/  /, ' '); // double spaces
263   				var a = str.split(/ /);
264   				$.each(a, function(i, val){
265   					if(val.substr(0,1) == '#'){ start = val.substr(1,val.length); }
266   					else if(val.substr(0,1) == '|'){ duration = val.substr(1,val.length); }
267   					else if(val.match(/(?=.ogg)/)){ url = val; }
268   					else if(val.length > 0){ id = val; }
269   				})
270   				//alert('   url:'+url +' start:'+ start +'  duration:'+ duration +'  id:'+ id);
271   				// build arr
272   				arr['id'] = id;
273   				arr['url'] = url;
274   				arr['seek'] = start == undefined ? 0 : start;
275   				arr['duration'] = duration == undefined ? 0 : duration;
276   				arr['annotation'] = [];
277   			return arr;
278   		},

281   		parseWikiHyperlink : function(str){ 
282   				var _this = this;
283   				var re = "";
284   				var tmp = '';
285   				var obj = {};
286   				obj.type = 'xlink';
287   				obj.linktype = 'standard';
288   	
289   				// link types // ?=.ogg | ?=.ogv | 
290   				re = new RegExp(/^\[\[http:\/\//);
291 					if(str.match(re)){ 
292 						// external link
293   					re = new RegExp(/\[\[http:\/\/[a-z A-Z 0-9 \#\ \_\/:.-]+/);
294   					tmp = (new String(re.exec(str))).split(" ");
295 						obj.target = tmp[0].replace(/^\[\[/,'');
296 						obj.title = tmp.length >= 2 ? tmp.slice(1) : tmp[0].replace(/^[\[\[http:\/\/]/, '');
297 						obj.title = (new String(obj.title)).replace(/,/g,' ');
298   					//alert(obj.target+' - '+obj.title);
299 						obj.linktype = 'external'; 
300 					}else{
301 						// standard links
302   					re = new RegExp(/\[\[[a-z A-Z 0-9 \# \ \_\/\|:.-]+/);
303   					tmp = (new String(re.exec(str))).split(/\|/);
304 						obj.target = tmp[0].replace(/^\[\[/,'');
305 						obj.title = tmp.length >= 2 ? tmp[1] : tmp[0].replace(/^\[\[/,'');
306   					//alert(obj.target+' - '+obj.title);
307 					}
308 					//alert(obj.target +'  '+ obj.linktype);						
309   				//alert(tmp.length+'  '+obj.title);
310   		
311 					// strip start time and duration
312 					var str2 = str.split(/\]/);
313 					re = new RegExp(/[\ ]\#[0-9]+/);  				
314   				obj.t1 = str.match(re) ? (new String(re.exec(str2[1]))).replace(/[\#]/, '') : 0; // .replace(/|\ /,'')
315 					re = new RegExp(/[\ ]\|[0-9]+/);  				
316   				obj.t2 = str.match(re) ? (new String(re.exec(str2[1]))).replace(/[\|]/, '') : 1000;
317 //					alert(obj.t1+' - '+obj.t2);
318 					
319 					// relative width/height: 50% 20%
320 					re = new RegExp(/[\ ]+[0-9]{2}(?=\%)/g);
321 					tmp = str.match(re) ? new String(str.match(re)).split(/,/) : [50,50];
322 					obj.x = tmp[0] ? tmp[0] : 50;
323 					obj.y = tmp[1] ? tmp[1] : 50;			
324 					//alert(''+obj.x+' - '+obj.y+'   time: '+obj.t1+' - '+obj.t2);
325 					
326 					 return obj;
327   		}
