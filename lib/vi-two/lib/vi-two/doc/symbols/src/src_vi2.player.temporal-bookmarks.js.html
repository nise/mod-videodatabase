 14 Vi2.TemporalBookmarks = $.inherit(/** @lends Vi2.TemporalBookmarks# */{ // 
 15 
 16 		/** @constructs
 17 		*		@param {object} options An object containing the parameters
 18 		*		@param {boolean} options.hasTimelineMarker Whether the TOC should be annotated on the timeline or not.
 19 		*		
 20 		*/
 21   	__constructor : function(options) { 
 22   			this.options = $.extend(this.options, options);  
 23 		},
 24 		
 25 		name : 'temporalBookmarks',
 26 		type : 'player-widget',
 27 		options : {
 28 			selector : '.control-bar',
 29 			label : 'bookmark'
 30 		},
 31 
 32 		/**
 33 		* Initializes the bookmark of content and handles options
 34 		*/
 35 		init : function(){ 
 36 			// get the url
 37 			var url = window.location.href.slice(window.location.href.indexOf('#') + 1) + '#!t=npt:'  ;
 38 			
 39 			// get the title of the video from database
 40 			var title = vi2.db.getMetadataById( vi2.observer.current_stream ).title;
 41 			
 42 			// clear selector
 43 			$( this.options.selector + '> .vi2-bookmark-controls' ).remove();
 44 		
 45 			var ico = $('<span></span>').addClass('glyphicon glyphicon-share')
 46 			// add button to player control bar
 47 			var container = $('<div></div>')
 48 				.html( ico )
 49 				//.append($('<div></div>').text( this.options.label ).addClass('vi2-bookmark-label'))
 50 				.addClass('vi2-bookmark-controls')
 51 				.bind('mouseenter', function(e){
 52 					$('.vi2-select-bookmark > input').val( url + vi2.observer.player.currentTime() ); 
 53 				})/*
 54 				.bind('mouseleave', function(e){
 55 					$('.bookmark-controls > .select-bookmark').css('display','none');
 56 				})*/
 57 				.appendTo( this.options.selector );
 58 			
 59 			var options = $('<span></span>')
 60 				.append( browserBookmark )			
 61 				.addClass('vi2-select-bookmark')
 62 				.appendTo( container );
 63 			
 64 							
 65 			var browserBookmark = $('<a></a>')
 66 				.text('add to browser')
 67 				.addClass('vi2-btn vi2-bookmarks-add-to-browser')
 68 				.attr('rel', 'sidebar')
 69 				.attr('href','#')
 70 				.attr('title', title)
 71 				.click(function() {
 72 					vi2.observer.log({context:'temporalBookmarks', action:'save-browser-bookmark',values:[''+url+vi2.observer.player.currentTime()]});
 73 				  if (window.sidebar) { // Mozilla Firefox Bookmark
 74 				    window.sidebar.addPanel(location.href,document.title,"");
 75 				  }else if(window.external) { // IE Favorite
 76 				    window.external.AddFavorite(location.href,document.title); 
 77 				  }else if(window.opera && window.print) { // Opera Hotlist
 78 				    this.title=document.title;
 79 				    return true;
 80 		 		 	}
 81 		 		 	return true;
 82 	 			})
 83 	 			.appendTo( options );	
 84 	 			
 85 	 		var input = $('<input type="text" />')
 86 				.val( url )
 87 				.attr('readonly',true)
 88 				.attr('aria-describedby', 'URL to the current playback position of the video.')
 89 				.click(function(){
 90 					vi2.observer.log({context:'temporalBookmarks', action:'copy-bookmark-link',values:[ ''+url + vi2.observer.player.currentTime()]})
 91 					$(this).focus();
 92 				})
 93 				.focus(function() { 
 94 					$(this).select(); 
 95 				} )
 96 				.appendTo( options );
 97 			
 98 		}
 99 }); // end class
