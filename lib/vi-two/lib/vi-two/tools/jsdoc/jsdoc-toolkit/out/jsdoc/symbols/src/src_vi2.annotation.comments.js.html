 26 		name : 'comments',
 27 		type : 'annotation',
 28 		options : {
 29 			hasTimelineMarker: true,

 42 		init : function(ann){
 43 			if( ann === null ){
 44 				ann = {};
 45 			} 
 46 			var _this = this;
 47 			var events = [];
 48 			$.each(ann, function(i, val){ 
 49 				if( val.type === _this.name ){  
 50 					events.push({
 51 						name: val.title, 
 52 						occ:[val.t1], 
 53 						time :[val.t1], 
 54 						date: val.date, 
 55 						author: val.author 
 56 					}); 
 57 				}
 58 			});
 59 			
 60 			// show comments in a menu
 61 			if( this.options.hasMenu ){
 62 				this.createMenu(events);
 63 			}
 64 			
 65 			// map events on the timeline
 66 			if( this.options.hasTimelineMarker ){ 
 67 				vi2.observer.player.timeline.addTimelineMarkers( 'comments', events, this.options.timelineSelector );
 68 			}				
 69 		},

 75 		createMenu : function(commentData){
 76 			var _this = this;
 77 			var tmp_t = -1;
 78 			
 79 			var comments = $('<ul></ul>')
 80 				.addClass('comments-list');
 81 			$( this.options.menuSelector ).html( comments );
 82 			
 83 			commentData = commentData.sort(function(a, b) {
 84   			return Number(a.time) > Number(b.time) ? 1 : -1;
 85 			});
 86 			moment.locale('de');
 87 			$.each( commentData, function(i, val){   
 88 			
 89 				var a = $('<a></a>')
 90 					.text(val.name)
 91 					.addClass('id-'+ val.time+' comments-menu-question' )
 92 					//.attr('href', '#'+vi2.options.id)
 93 					.click(function(){  
 94 						vi2.observer.log({ context:'comments',action:'menu-click',values: [val.name, val.author, val.time[0] ]} );
 95 						_this.player.currentTime( val.time[0] );
 96 					})
 97 					;	
 98 					
 99 				if( _this.options.allowEmoticons ){
100 					a.emoticonize({ /* delay: 800, animate: false, exclude: 'pre, code, .no-emoticons' */ });
101 				}	
102 				
103 				
104 				var user = vi2.db.getUserById( val.author );	
105 				
106 				var header = $('<span></span>')
107 					.addClass('comments-header');
108 					
109 				$('<span></span>')
110 					//.text( user.firstname +' '+user.name )
111 					.text( user.username )
112 					.addClass('comments-user')
113 					.appendTo( header )
114 					; 
115 				header.append(' ' + moment(Number(val.date), "x").fromNow() );	
116 				
117 				var li = $('<li></li>')
118 					.addClass('list-item')
119 					.addClass('t'+val.time)
120 					.attr('author', val.author)
121 					.attr('date', 'd'+val.date)
122 					.attr('id', 't'+_this.formatTime(val.time, '-'))
123 					.html( header )
124 					.append(a)
125 					.appendTo(comments)
126 					;
127 
128 				if( Number(val.time) === Number(tmp_t) ){ 
129 					li.css({ 'margin-left':'15px' }); 
130 					// re-comments could be sorted desc by date. Solution needed
131 					//comments.find('.t'+val.time).tsort({ attr:"date", order:'asc'}); 
132 				}	
133 
134 				// edit
135 				if( _this.options.allowEditing && Number(val.author) === Number(vi2.wp_user) ){	 
136 					var edit_btn = $('<a></a>')
137 						.addClass('tiny-edit-btn glyphicon glyphicon-pencil' )
138 						.attr('data-toggle', "modal")
139 						.attr('data-target', "#myModal")
140 						.attr('data-annotationtype', 'comments')
141 						.data('annotationdata', { 
142 							content: val.name, 
143 							time: val.time, 
144 							date: val.date,
145 							author: val.author 
146 						} )
147 						.appendTo( header )
148 						;
149 				}		
150 				// re-comments
151 				if( _this.options.allowReplies ){		
152 					var reply_btn = $('<a></a>')
153 						.addClass('tiny-edit-btn glyphicon glyphicon-arrow-right' )
154 						.attr('data-toggle', "modal")
155 						.attr('data-target', "#myModal")
156 						.attr('data-annotationtype', 'comments')
157 						.data('annotationdata', { content: '', time: val.time, date: (new Date().getTime()) } )
158 						.appendTo( header )
159 						;		
160 					}	
161 					
162 				tmp_t = val.time;		
163 			}); // end each
164 		},

169 		appendToDOM : function(id){ 
170 			var _this = this;
171 			$(vi2.dom).find('[type="comments"]').each(function(i,val){ $(this).remove(); });
172 			$.each(	vi2.db.getCommentsById(id), function( i, val ){ 
173 				var comm = $('<div></div>')
174 					.attr('type',"comments")
175 					.attr('starttime', val.start)
176 					.attr('duration', 10)
177 					.attr('author', val.author)
178 					.attr('date', val.date)
179 					.text( decodeURIComponent(val.comment))
180 					.appendTo( vi2.dom )
181 					;  
182 			});
183 		},
