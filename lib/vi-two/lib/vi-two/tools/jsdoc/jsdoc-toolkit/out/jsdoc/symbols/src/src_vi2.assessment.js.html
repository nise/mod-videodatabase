167 		evaluateAnswer : function(question_selector, obj){
168 			if(obj.answ[0].questiontype == 'fill-in'){
169 				$(question_selector).append($('<p>Vielen Dank für die Bearbeitung der Frage.</p>').addClass('assessment-msg-correct'));
170 			}else{ 
171 				var one_checked = false, correct = true;
172 				obj.checked = [];
173 				$('.assessment-msg-warning').hide();
174 				$('.assessment-answers').find('div.assessment-answer').each(function(i, val){
175 					if($(this).find("input[name='quest']:checked").val() == 1){
176 						obj.checked.push($(this).text())
177 						one_checked = true;
178 						$.each(obj.correct, function(j, corr){ 
179 							if(corr == $(val).attr('id')){	
180 								obj.correct[j] = true; 
181 							}
182 						});
183 					}
184 				});
185 			
186 				// VALIDATION
187 				if( ! one_checked ){
188 					//vi2.observer.log('[call:run_assessment, result:empty_selection]'); 
189 					//alert('pls select one');
190 					$(question_selector).append($('<p>Bitte wählen Sie eine Antwortoption</p>').addClass('assessment-msg-warning'));
191 					$('.assessment-btn').show();
192 					return false;
193 				}
194 			
195 				// CORRECT?
196 				$.each(obj.correct, function(i, val){ 
197 					if( val != true ) {
198 						vi2.observer.log('assessmentwrong');
199 						correct = false;
200 					}
201 				});
202 				if(correct == true){
203 					vi2.observer.log('assessmentcorrect');
204 					$(question_selector).append($('<p>Ihre Antwort ist richtig.</p>').addClass('assessment-msg-correct'));
205 					//vi2.observer.log('[call:run_assessment, result:correct]');
206 				}else{ // wrong
207 					//vi2.observer.log('[call:run_assessment, result:wrong]');
208 					 
209 					$(question_selector).append($('<p>Ihre Antwort ist leider falsch.</p>').addClass('assessment-msg-wrong'));
210 				}
211 			}
212 			// save result to node
213 			obj.type = obj.answ[0].questiontype;
214 			var result = {
215 				correct: correct,
216 				question : encodeURIComponent(obj.question),
217 				answ : obj.answ,
218 				res : obj.type == 'mc' ? encodeURIComponent(obj.checked) : encodeURIComponent($('.assessment-answers').find('textarea').val()),
219 				videoid : vi2.currentVideo
220 			}; 
221 			var question_result = {
222 		  	from : vi2.wp_user, // the user that answered the question
223 		  	to : obj.author, // the author of the question
224 		  	date : (new Date()).getTime(),
225 		  	type : 'test-result',
226 		  	read : false, 
227 		  	replied: false,
228 		  	title : 'Result: '+encodeURIComponent(obj.question),
229 		  	content : result 
230     	};
231 			$.post('/messages', {"data":question_result}, function(res){ 
232         //alert('Has been saved: '+ JSON.stringify(res)); 
233     	});

285 		formatTime : function(secs){
286 			var seconds = Math.round(secs);
287     	var minutes = Math.floor(seconds / 60);
288     	minutes = (minutes >= 10) ? minutes : "0" + minutes;
289     	seconds = Math.floor(seconds % 60);
290     	seconds = (seconds >= 10) ? seconds : "0" + seconds;
291     	return minutes + "-" + seconds;
292 		}
