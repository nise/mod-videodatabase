 12 (function($) {
 13 
 14 var hasIntrospection = (function(){_}).toString().indexOf('_') > -1,

 17 $.inherit = function() {
 18 
 19 	var withMixins = $.isArray(arguments[0]),
 20 		hasBase = $.isFunction(arguments[0]) || withMixins,
 21 		base = hasBase? withMixins? arguments[0][0] : arguments[0] : emptyBase,
 22 		props = arguments[hasBase? 1 : 0] || {},
 23 		staticProps = arguments[hasBase? 2 : 1],
 24 		result = props.__constructor || base.prototype.__constructor?
 25 			function() {
 26 				this.__constructor.apply(this, arguments);
 27 			} : function() {};
 28 
 29 	if(!hasBase) {
 30 		result.prototype = props;
 31 		result.prototype.__self = result.prototype.constructor = result;
 32 		return $.extend(result, staticProps);
 33 	}
 34 
 35 	var inheritance = function() {};
 36 
 37 	$.extend(result, base, staticProps);
 38 
 39 	inheritance.prototype = base.prototype;
 40 	result.prototype = new inheritance();
 41     var resultPtp = result.prototype;
 42 	resultPtp.__self = resultPtp.constructor = result;
 43 
 44 	var propList = [];
 45 	$.each(props, function(i) {
 46 		props.hasOwnProperty(i) && propList.push(i);
 47 	});
 48 	// fucking ie hasn't toString, valueOf in for
 49 	$.each(['toString', 'valueOf'], function() {
 50 		props.hasOwnProperty(this) && $.inArray(this, propList) == -1 && propList.push(this);
 51 	});
 52 
 53 	var basePtp = base.prototype;
 54 	$.each(propList, function() {
 55 		if(hasBase
 56 			&& $.isFunction(basePtp[this]) && $.isFunction(props[this])
 57 			&& (!hasIntrospection || props[this].toString().indexOf('.__base') > -1)) {
 58 
 59 			(function(methodName) {
 60 				var baseMethod = basePtp[methodName],
 61 					overrideMethod = props[methodName];
 62 				resultPtp[methodName] = function() {
 63 					var baseSaved = this.__base;
 64 					this.__base = baseMethod;
 65 					var result = overrideMethod.apply(this, arguments);
 66 					this.__base = baseSaved;
 67 					return result;
 68 				};
 69 			})(this);
 70 
 71 		}
 72 		else {
 73 			resultPtp[this] = props[this];
 74 		}
 75 	});
 76 
 77 	if(withMixins) {
 78 		var i = 1, mixins = arguments[0], mixin, __constructors = [];
 79 		while(mixin = mixins[i++]) {
 80 			$.each(mixin.prototype, function(propName) {
 81 				if(propName == '__constructor') {
 82 					__constructors.push(this);
 83 				}
 84 				else if(propName != '__self') {
 85 					resultPtp[propName] = this;
 86 				}
 87 			});
 88 		}
 89 		if(__constructors.length > 0) {
 90 			resultPtp.__constructor && __constructors.push(resultPtp.__constructor);
 91 			resultPtp.__constructor = function() {
 92 				var i = 0, __constructor;
 93 				while(__constructor = __constructors[i++]) {
 94 					__constructor.apply(this, arguments);
 95 				}
 96 			};
 97 		}
 98 	}
 99 
100 	return result;

102 };
