128 	json_import : function(){ 
129 		var _this = this;
130 		var lectures = '';
131 		var images = ''; 
132 		$.ajax({
133     	type: "POST",
134     	dataType: "json",
135     	url: './data_slide_update.json',
136     	success: function(lec){   
137     		
138     		$.each(lec.data, function(i, val){
139     			images = '';//val.title+"\n";	
140     				 
141     			$.each(val.project.media, function(ii, media){	
142     				$.each(media.tracks, function(i, track){   
143     					$.each(track.trackEvents, function(i, img){    
144     						if(img.type === 'image'){
145     							images += '{ "type":"seq", "starttime":'+img.popcornOptions.start+', "duration":'+(img.popcornOptions.end - img.popcornOptions.start)+', "id":"", "img":"'+String(img.popcornOptions.src).replace("http://elearning.ihi-zittau.de/beta/iwrm/slides/"+val.title+"/", "")+'" },';	
146 								}else if(img.type === 'pop'){
147 									images += '#x: '+img.popcornOptions.left+'  #y: '+img.popcornOptions.top+'  #start: '+_this.seconds2deci(img.popcornOptions.start)+'  #dur: '+(img.popcornOptions.end-img.popcornOptions.start)+'  #text: '+img.popcornOptions.text+'  #target: '+img.popcornOptions.link;
148     							images += "\n";
149     						}  				
150     					});
151     				});
152     			});	
153     			lectures += '\n { "id": "'+val.title+'", "slides":['+ images.substr(0, images.length - 1) +']},\n';			
154     		});
155     		
156     		var jsoon = '';
157 				jsoon += '{ "_name": "vi2 slides", "_slides": [\n';
158 				jsoon += lectures.substr(0, lectures.length -1 );
159 				jsoon += '\n]}';
160 				
161 				$('#debug').html(jsoon);
162     	},
163 			error: function(e, msg, x){ alert('error at json import '+msg); },
164 		});
