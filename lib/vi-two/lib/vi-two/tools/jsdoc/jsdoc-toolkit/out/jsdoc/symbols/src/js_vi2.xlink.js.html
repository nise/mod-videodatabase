 17 	var XLink = $.inherit(/** @lends XLink# */{
 18 
 19 		/** @constructs
 20 		*		@extends Annotation
 21 		*		@param {object} options An object containing the parameters
 22     *		@param {string} options.target_selector A required setting.
 23     *		@param {string} options.selector An optional setting.
 24 		*/
 25   	__constructor : function(options) {
 26   		this.options = $.extend(this.options, options); 
 27   		this.observer = observer;
 28 		},
 29 				
 30 		name : 'xlink',
 31 		type : 'annotation',
 32 		observer : undefined,
 33 		
 34 		options : {target_selector:'#seq' ,selector: '#overlay', vizOnTimeline: true, minDuration: 5},
 35 		player : null,
 36 		timelineSelector : 'div.vi2-video-seek', 
 37 		link_list : {},
 38 		currLinkId :-1,
 39 
 40 		/* ... */
 41 		init : function(ann){
 42 			var _this = this;
 43 			this.link_list = this.buildLinkList(ann);	
 44 			this.showTimelineXlink(this.link_list);
 45 		},		
 46 		
 47 		/* Translated database entry of link into a dom element that the parser will read later on */
 48 		// // <div type="xlink" starttime=297 duration=14 posx=32 posy=90 id="Using existing Videos" >bonk1</div>
 49 		appendToDOM : function(id){
 50 			var _this = this;
 51 			$.each(	vi2.db.getLinksById(id), function(val){ 
 52 				var links = $('<div></div>')
 53 				.attr('type', this.type) // former default: "xlink"
 54 				.attr('starttime', deci2seconds(this.start))
 55 				.attr('duration', this.duration)
 56 				.attr('posx', this.x)
 57 				.attr('posy', this.y)
 58 				.attr('seek', deci2seconds(this.seek))
 59 				.attr('duration2', deci2seconds(this.duration2))
 60 				.attr('id', this.id)
 61 				.text('#!'+this.target)
 62 				.appendTo( vi2.dom )
 63 				;
 64 			});
 65 		},				
 66 		
 67 		
 68 		/* ... */
 69 		buildLinkList : function(ann){
 70 			var e = {}; e.tags = []; e.tags.occ = [];
 71 			$.each(ann, function(i, val){
 72 				if(val.linktype == 'cycle' || val.linktype == 'standard' || val.linktype == 'xlink'){ 
 73 					e.tags.push({name: val.title, occ:[val.t1], target:val.target});
 74 				}
 75 			});
 76 			return e;
 77 		},
 78 
 79 		
 80 		/** Begin of XLink annotation. Typically the link anchor will apeare on screen. There are three different link types: standard (target within video collection), external (target elsewhere in the WWW) and cycle (like standard link but with the option to return to the link source).
 81 				@param {Object} e
 82 				@param {String} id
 83 				@param {Object} obj		
 84 		*/
 85 		begin : function(e, id, obj){ 
 86 				this.currLinkId = id;
 87 				var _this = this;
 88 				var pos = this.relativePos(obj.displayPosition); 
 89 				var o = $('<a></a>')
 90 					.text(obj.content.title)
 91 					//.attr('href', obj.content.target)
 92 					.attr('id', 'ov'+id)
 93 					.addClass('overlay ov-'+id)
 94 					.addClass('hyperlink-'+obj.linktype)
 95 	 				.bind('click', {}, function(data){
 96 	 					// distinguish link types
 97 	 					switch(obj.linktype){
 98 	 						case 'standard' : 	// called xlink
 99 	 							var new_stream = obj.content.target.replace(/\#!/,'');
100 								vi2.observer.setCurrentStream(new_stream);
101 								break;
102 							case 'external' :
103 								return true;
104 							case 'cycle' : 
105 								var new_stream = obj.content.target.replace(/\#!/,'');
106 								vi2.observer.setCurrentStream(new_stream);
107 								// make new object for return link
108 								var return_obj = {
109 									title : 'return to: '+obj.content.title,
110 									target : String(_this.player.url).replace(/.webm/,'').replace(/videos\/iwrm\_/,''), // dirty IWAS hack
111 									linktype : 'standard',
112 									type : 'xlink',
113 									x : 2, //obj.displayPosition.x,
114 									y : 93, // obj.displayPosition.y,
115 									t1 : obj.seek,
116 									t2 : obj.duration,
117 									seek : obj.displayPosition.t1,
118 									duration : 0
119 								};	
120 								// append that object
121 								vi2.observer.vid_arr[0]['annotation'].push(return_obj); 
122 								//	_this.loadCycleVideo(obj.content.target, 10, 15, obj.displayPosition.t1); // url, seek time, duration, return_seek
123 								break;
124 							case 'x':
125 								break;	
126 						}
127 						// load Video
128 						_this.loadVideo(vi2.observer.vid_arr[0]['url'], obj.seek);
129 						// log something
130 						vi2.observer.log(_this.player.url+' link_click: '+obj.content.target+', seek_to: '+obj.displayPosition.t1);
131 						// remove link ancshor after click 
132 						$(this).remove();
133 					});
134 				$(this.options.selector).append(o);
135 				// positioning object AFTER appending it to its parent // buggy
136 				// ($(this.options.selector).offset()).left+
137 				//alert($(this.options.selector).offset().left+'  '+pos.x);
138 				o.css({left: pos.x, top: pos.y, position:'absolute'});		
139 		//	}	
140 		},
141 	
142 		/* End of annotion time. The link anchor will disapear from screen. */
143 		end : function(e, id){ 	 //alert('end link')
144 			$(this.options.selector+' .ov-'+id).hide();
145 		},
146 		
147 		/* Visualizes link representations on the timeline */
148 		showTimelineXlink : function(e){
149 				var _this= this; 
150 				$(_this.timelineSelector+' .timetag').remove();
151 				// display tag occurence on timeline to motivate further selection
152 				var f = function(_left, _name, start){
153 					var sp = $('<span></span>');					
154 					sp.addClass('timetag tlink').attr('style','left:'+_left+'px;')
155 						.tooltip({delay: 0, showURL: false, bodyHandler: function() { return $('<span></span>').text('Link to: '+_name);} })
156 						.bind('click', function(event){
157 							_this.player.currentTime(start);
158 							vi2.observer.log(_this.player.url+' timeline_link_seek: '+start);
159 						});
160 					return sp;
161 				};
162 				//				
163 				$.each(e.tags, function(){ 					
164 					var progress = this.occ[0] / _this.player.duration();
165 					progress = ((progress) * $(_this.timelineSelector).width()); 
166   	    	if (isNaN(progress) || progress > $(_this.timelineSelector).width()) { return;}
167 	 				$(_this.timelineSelector+'link').append(f(progress, this.name, this.occ[0]));
168  				});
169 		},

189 		relativePos : function(obj){ 
190 			//var pplayer = observer.widget_list['seq']; // IWRM only fix xxx // bugy
191 			//return {x: Math.floor((obj.x/100)*pplayer.width()), y: Math.floor((obj.y/100)*pplayer.height())};
192 			return {x: Math.floor((obj.x/100)*615), y: Math.floor((obj.y/100)*450)};
193 		},
