 29 					case 'html' :

 35 			parseWiki : function(){ 
 36 				var _this = this;
 37 			  var v_id = -1;
 38 			  // dirty hack for mediawiki xxx
 39 			  $(this.selector).val($(this.selector).val().replace(/\<p\>/, ''));
 40 				// go through markup lines
 41   			$($(this.selector+' > p').text().split(/\n/)).each(function(i, val){ 
 42   				if(val.substr(0,8) == "[[Video:" || val.substr(0,8) == "[[video:"){ 
 43 						// parse videos to sequence
 44 						v_id++; 	  				  		
 45   					_this.vid_arr[v_id] = _this.parseWikiVideo(val);
 46 	  				}else if(val.substr(0,2) == "[["){
 47 	  				// parse hyperlinks related to the latter video 
 48   					_this.vid_arr[v_id]['annotation'].push(_this.parseWikiHyperlink(val)); //alert('ok_'+val);  					
 49   				}else{
 50   					//alert('bug_'+val);
 51   				}
 52   			}); 			
 53 				return this.vid_arr;  			
 54 			},

 58 				var _this = this;
 59 			  var v_id = -1;
 60 			  var arr = [];
 61 			  var obj = {};
 62   			$('div'+this.selector+' div').each(function(i, val){ 
 63   				if($(this).attr('type') == "video"){

 65   					arr = [];

 72   					_this.vid_arr[v_id] = arr;

 75   					// standard and external links
 76   					obj = {};

 89 						if(obj.target.match(/(^http:)/)){ obj.linktype = 'external'; } // external link
 90   					
 91   					
 92   					_this.vid_arr[v_id]['annotation'].push(obj);

153   		/* ... */ // all of that is quick & dirty and needs further testing / testing procedures
154   		parseWikiVideo : function(str){
155 
156 					var arr = [];
157 					var url, start, duration, id = '_';
158   				str = str
159   					.replace(/^\[\[Video:/, '') // start delimiter
160   					.replace(/\]\]/, '') // end delimiter
161   					.replace(/\# /, ' #') // start-time
162   					.replace(/\| /, ' |') // duration
163   					.replace(/  /, ' '); // double spaces
164   				var a = str.split(/ /);
165   				$.each(a, function(i, val){
166   					if(val.substr(0,1) == '#'){ start = val.substr(1,val.length); }
167   					else if(val.substr(0,1) == '|'){ duration = val.substr(1,val.length); }
168   					else if(val.match(/(?=.ogg)/)){ url = val; }
169   					else if(val.length > 0){ id = val; }
170   				})
171   				//alert('   url:'+url +' start:'+ start +'  duration:'+ duration +'  id:'+ id);
172   				// build arr
173   				arr['id'] = id;
174   				arr['url'] = url;
175   				arr['seek'] = start == undefined ? 0 : start;
176   				arr['duration'] = duration == undefined ? 0 : duration;
177   				arr['annotation'] = [];
178   			return arr;
179   		},

182   		parseWikiHyperlink : function(str){ 
183   				var _this = this;
184   				var re = "";
185   				var tmp = '';
186   				var obj = {};
187   				obj.type = 'xlink';
188   				obj.linktype = 'standard';
189   	
190   				// link types // ?=.ogg | ?=.ogv | 
191   				re = new RegExp(/^\[\[http:\/\//);
192 					if(str.match(re)){ 
193 						// external link
194   					re = new RegExp(/\[\[http:\/\/[a-z A-Z 0-9 \#\ \_\/:.-]+/);
195   					tmp = (new String(re.exec(str))).split(" ");
196 						obj.target = tmp[0].replace(/^\[\[/,'');
197 						obj.title = tmp.length >= 2 ? tmp.slice(1) : tmp[0].replace(/^[\[\[http:\/\/]/, '');
198 						obj.title = (new String(obj.title)).replace(/,/g,' ');
199   					//alert(obj.target+' - '+obj.title);
200 						obj.linktype = 'external'; 
201 					}else{
202 						// standard links
203   					re = new RegExp(/\[\[[a-z A-Z 0-9 \# \ \_\/\|:.-]+/);
204   					tmp = (new String(re.exec(str))).split(/\|/);
205 						obj.target = tmp[0].replace(/^\[\[/,'');
206 						obj.title = tmp.length >= 2 ? tmp[1] : tmp[0].replace(/^\[\[/,'');
207   					//alert(obj.target+' - '+obj.title);
208 					}
209 					//alert(obj.target +'  '+ obj.linktype);						
210   				//alert(tmp.length+'  '+obj.title);
211   		
212 					// strip start time and duration
213 					var str2 = str.split(/\]/);
214 					re = new RegExp(/[\ ]\#[0-9]+/);  				
215   				obj.t1 = str.match(re) ? (new String(re.exec(str2[1]))).replace(/[\#]/, '') : 0; // .replace(/|\ /,'')
216 					re = new RegExp(/[\ ]\|[0-9]+/);  				
217   				obj.t2 = str.match(re) ? (new String(re.exec(str2[1]))).replace(/[\|]/, '') : 1000;
218 //					alert(obj.t1+' - '+obj.t2);
219 					
220 					// relative width/height: 50% 20%
221 					re = new RegExp(/[\ ]+[0-9]{2}(?=\%)/g);
222 					tmp = str.match(re) ? new String(str.match(re)).split(/,/) : [50,50];
223 					obj.x = tmp[0] ? tmp[0] : 50;
224 					obj.y = tmp[1] ? tmp[1] : 50;			
225 					//alert(''+obj.x+' - '+obj.y+'   time: '+obj.t1+' - '+obj.t2);
226 					
227 					 return obj;
228   		}
