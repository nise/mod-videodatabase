  8 var Vi2_Playlist = $.inherit(/** @lends Playlist# */{
  9 
 10 	/** 
 11 	*		@constructs 
 12 	*		@param {object} options An object containing the parameters
 13 	*/
 14 	__constructor : function(options) {
 15 		this.options = $.extend(this.options, options);	 
 16 	},
 17 	
 18 	name : 'playlist',
 19 	options : {selector: '#null'},
 20 	content_selector : '#content',
 21 	playlist : [],
 22 	
 23 	
 24 	/* adds item to playlist */
 25 	add : function(id){
 26 		this.playlist.push(id);
 27 	},
 28 	
 29 	
 30 	/* removes item from playlist */
 31 	remove : function(id){
 32 		var _this = this;
 33 		$(this.playlist).each(function(i, val){
 34 			if(val == id){
 35 				_this.playlist.splice(i, 1);		
 36 			}
 37 		});
 38 	},
 39 	
 40 	
 41 	/* Renders playlist items regarding presentation template */
 42 	showPlaylist : function(){
 43 		var _this = this;
 44 		
 45 		// handle empty playlist
 46 		if(_this.playlist.length == 0){
 47 				$(_this.content_selector).append('(Playlist is empty)')
 48 		}
 49 		// remove duplicates
 50 		this.playlist = removeDuplicates(_this.playlist);
 51 				
 52 		$.each(this.playlist, function(i, list_item){ 
 53 			var stream = vi2.db.getStreamById(list_item)
 54 			var item =$('<div></div>')
 55 				.addClass('content-item')
 56 				.setTemplate($("#item_template").val())
 57 				.processTemplate(stream);
 58 			$(_this.content_selector).append(item);
 59 		});				
 60 	}
 61 	
 62 	
 63 	
 64 }); // end class
