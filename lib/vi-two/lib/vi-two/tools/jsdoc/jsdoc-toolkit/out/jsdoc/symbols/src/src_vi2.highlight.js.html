127 		init : function(ann){ 
128 			var _this = this;
129 			_this.tag_obj = [];
130 			$.each(ann, function(i, val){  
131 				if(val.type == 'highlight'){ 
132 					var ii = _this.get_tag_by_name(val.title);
133 					if( ii == -1){ 
134 						_this.tag_obj.push({tagname: val.title, occ: [{start: val.t1, duration: val.t2, xpos:val.x, ypos:val.y}]});
135 					}	else {
136 					_this.tag_obj[ii].occ.push({start: val.t1, duration: val.t2, x:val.x, y:val.y});
137 					}
138 				}
139  			}); 
140 			this.displayTagcloud(); 
141 		},

147 			var _this = this; 
148 			//$(vi2.dom).find('[type="tags"]').each(function(i,val){ $(this).remove(); }); 
149 			
150 			$.each(	vi2.db.getHighlightById(id), function( i, val ){   
151 				$.each(val.occ, function(j, occurance){		
152 					var toc = $('<div></div>')
153 						.attr('type',"highlight")
154 						.attr('starttime', occurance.start)
155 						.attr('id', "tag-"+j)	
156 						.attr('posx', occurance.x)
157 						.attr('posy', occurance.y)
158 						.attr('duration', occurance.duration)

167 		get_tag_by_name : function(name){
168 			var out = -1;
169 			$.each(this.tag_obj, function(i, val){
170 				if(val.tagname == name){
171 					out = i;
172 				} 
173 			});
174 			return out;
175 		},

178 		displayTagcloud : function(){  
179 			$(this.options.selector).empty();
180 			var _this = this;
181 			// template for displaying tags
182 			
183 			// prepare list and append existing tags
184 			var ul = $("<ul></ul>").addClass("highlight-list").appendTo(this.options.selector);   
185 			$.each(this.tag_obj, function(i, val){
186 				var li = $('<li></li>')
187 					.addClass('list-item')
188 					.attr('id', val.tagname.replace(' ', '--'))
189 					.append($('<a></a>')
190 					.attr('freq', val.occ.length )
191 					.addClass('id-'+val.tagname.replace(' ', '--'))
192 					.text(val.tagname+', ')//+'('+val.occ.length+') ')
193 					.css("font-size", (val.occ.length / 10 < 1) ? val.occ.length / 10 + 1 + "em": (val.occ.length / 10 > 2) ? "2em" : _freq / 10 + "em")
194 					.bind('click', {tags: val }, function(e){
195 						$('.highlight-element').remove(); 
196 						_this.showTimelineTag(e.data);
197 						vi2.observer.log('clicktagfromlist:'+ val.tagname); 
198 					})
199 				).appendTo( ul ); 
200 			});
201 			// sort by occurence or alphabeticly, sort order desc / asc
202 			var sortAttr = {};
203 			sortAttr.attr = this.options.sort == 'freq' ? 'freq' : '';
204 			sortAttr.order = this.options.order == 'desc' ? "desc" : "asc";
205 			ul.find('a').tsort(sortAttr); 
206 			// cut off elements above max and render them
207 			$(ul).find('li:gt('+(this.options.max-1)+') > a').hide();
208 			
209 		},

212 		showTimelineTag : function(e){  
213 			var _this = this; 
214 			$(_this.options.timeline_selector+' .ttag').remove();
215 			if(e.tags.occ.length === 1 && e.tags.occ[0].start >= 0){ 
216 				// jump to temporal position 
217 				this.player.currentTime(e.tags.occ[0].start); 
218 				// highlight object inside the video 
219 				this.placeHighlight({}, 2,{ content: { title: e.tags.tagname}, displayPosition: {x: e.tags.occ[0].xpos, y: e.tags.occ[0].ypos, width: '100px', height:'100px'}});
220 			}else{ 
221 				// display tag occurence on timeline to motivate further selection
222 				$.each(e.tags.occ, function(){ 
223 					var progress = this.start / _this.player.duration();
224 					progress = ((progress) * $(_this.options.timeline_selector).width());
225   	    	if (isNaN(progress)) { progress = 0; }
226 					var span = $('<span></span>')
227 						.addClass('highlight-timeline-entry')
228 						.attr('style','left:'+ progress +'px;')
229 						.tooltip({delay: 2, showURL: false, bodyHandler: function() { return $('<span></span>').text(''+ e.tags.tagname);} })				
230 						.bind('click', function(){
231 							_this.player.currentTime( this.start );
232 							// highlight object inside the video
233 							
234 						})
235 						;
236 						/*.bind('mouseover', function(){
237 							...tooltip  e.tags.tagname
238 						});*/
239 	 				$(_this.options.timeline_selector).append( span );
240  				});
241 			}
242 		},

246 			*

249 		placeHighlight: function(e, id, obj){
250 			var _this = this; 
251 				var pos = this.relativePos(obj.displayPosition); 
252 				var o = $('<a></a>')
253 					//.text(''+obj.tagname )
254 					//.attr('href', obj.content.target)
255 					.attr('id', 'ov'+id)
256 					.addClass('overlay ov-'+id)
257 					.addClass('highlight-element')
258 	 				.bind('click', {}, function(data){
259 	 					_this.showOverlay( String(obj.content.title).replace(/\ /g, '_') );
260 					})
261 					.bind('mouseover', {}, function(e){
262 						$(this).text(obj.content.title);
263 					})
264 					.bind('mouseout', {}, function(e){
265 						$(this).text('');
266 					})
267 				// append and position object
268 				$(this.options.overlay_selector).append(o); 
269 				o.css({ 
270 					left: pos.x, 
271 					top: pos.y, 
272 					position:'absolute', 
273 					width: obj.displayPosition.width, 
274 					height: obj.displayPosition.height 
275 				})
276 				.show();
277 			
278 			// remove highlight layer when playback starts
279 			$( vi2.observer.player ).on('player.play', function(e){
280 				$('.highlight-element').remove();
281 				_this.popup.dialog( "close" );
282 			});		
283 		}
