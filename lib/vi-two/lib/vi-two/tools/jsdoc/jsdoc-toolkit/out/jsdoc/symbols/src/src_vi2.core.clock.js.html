 31 Vi2.Clock = $.inherit(/** @lends vi2.core.Clock# */
 32 	{
 33 			/** 
 34 			*		@constructs 
 35 			*		@param {Videoplayer} player Related video player 
 36 			*		@param {Number} clockInterval Interval of clock granularity in milliseconds
 37 			*/
 38   		__constructor : function(player, clockInterval) {
 39   			this.player = player;
 40   			this.clockInterval = clockInterval;  		
 41   		},
 42   		
 43 		
 44 		name : 'clock',
 45 		player : null,
 46 		clockInterval : 500, // = default
 47 		isRunning : false,
 48 		timelineSelector : 'div.vi2-video-seek',
 49 		interval : -1,	
 50 		annotations : [],
 51 		hooks : [],
 52 		
 53 		/**
 54 		* Checks whether the provided type of annotation has been already added as a hook
 55 		* @param {String} type
 56 		* @return {boolean}
 57 		*/
 58   	isHook : function(type){
 59   		return this.hooks[type] !== null;	
 60   	},
 61   	
 62   	/* ... */
 63   	addHook : function(type, fn){
 64   		this.hooks[type] = fn;
 65   		return true;
 66   	},
 67 	
 68 		/* push annotation on their stack by mapping the parser object to the specific annotation object structure */
 69 		addAnnotation : function(obj){ 					
 70 			if(this.isHook(obj.type)){   
 71 				this.annotations.push({
 72 						active:false,
 73 						author: obj.author,
 74 						width: obj.width, 
 75 						content: {
 76 								title: obj.title, 
 77 								target:obj.target,
 78 								note:obj.note
 79 							}, 
 80 						linktype:obj.linktype, 
 81 						type: obj.type, 
 82 						displayPosition: {
 83 								x: obj.x, 
 84 								y: obj.y, 
 85 								t1: obj.t1, 
 86 								t2: obj.t2
 87 							},
 88 							seek : obj.seek,
 89 							duration : obj.duration
 90 						});
 91 		 	}	
 92 		},
 93 	
 94 		/* Trivial */
 95 		checkAnnotation : 	function() {
 96 			var iTime = this.parseTime( this.player.currentTime() );
 97 
 98 			for (var i=0; i < this.annotations.length;i++){
 99 				var oAnn = this.annotations[i];
100 				if(iTime >= oAnn.displayPosition.t1 && iTime < (Number(oAnn.displayPosition.t1) + Number(oAnn.displayPosition.t2))) {
101 					if(!oAnn.active){
102 						oAnn.active = true; 
103 	  				$(this.player).trigger('annotation.begin.'+oAnn.type, [i, oAnn]); 
104 					}
105 				}else {
106 					oAnn.active = false;
107 	  			$(this.player).trigger('annotation.end.'+oAnn.type, [i]);
108 				}
109 			}
110 		},
111 		
112 		
113 		
114 	/* Optimized algorithm, making advantage of indexing the time of appearance */
115 		/* to do: 
116 		-generate test data, 
117 		-try different approaches to generate the index, 
118 		-measure time, 
119 		-calculate complexity in landau notation
120 		*/
121 		prepAnno : [],
122 		
123 		buildAnnotationIndex : function(){
124 			var prepAnno = [];
125 			for (var i = 0; i < 1000; i++){ prepAnno[i] = [];} 
126 			$.each(this.annotations, function(i, val){
127 				var index = val.displayPosition.t1 < 1 ? 0 : Math.ceil(val.displayPosition.t1 / 100);
128 				prepAnno[index].push(val);
129 			});
130 			this.prepAnno = prepAnno;
131 		},
132 		
133 		checkAnnotation_new : 	function() {
134 			var _this = this;
135 			var iTime = this.parseTime( this.player.currentTime() ); // returns time in decimal 
136 			var x = this.player.currentTime() < 1 ? 0 : Math.ceil(iTime / 100); 
137 			$('#debug').val(_this.prepAnno[x].length);	
138 			$.each(_this.prepAnno[x], function(i, oAnn){
139 				
140 				if(iTime >= oAnn.displayPosition.t1 && iTime < (Number(oAnn.displayPosition.t1) + Number(oAnn.displayPosition.t2))) {
141 					if(!oAnn.active){
142 						oAnn.active = true; 
143 	  				$(_this.player).trigger('annotation.begin.'+oAnn.type, [i, oAnn]); 
144 					}
145 				}else {
146 					oAnn.active = false;
147 	  			$(_this.player).trigger('annotation.end.'+oAnn.type, [i]);
148 				}
149 			});
150 		},
151 		
152 		
153 		// Another approach
154 		/*
155 		1. generate an inverted index where Index[ playbacktime_rounded ] = {annotation_1, annotation_2, ..., annotation_n}. The Index contains all annotations, that should be visible at time. 
156 		2. at a given playbacktime just test whether the time in ms exists in the Index.
157 		
158 		Problem:: Size of Index
159 		*/
160 		
161 							
162 		/* ... */
163 		startClock : function(){  
164 			//this.buildAnnotationIndex();
165 			if(this.isRunning){ return;}
166 			var _this = this;
167 			this.isRunning = true;
168 			this.interval = setInterval(function() { _this.checkAnnotation();  }, this.clockInterval);		
169 		},
170 		
171 		/* ... */
172 		stopClock : function(){
173 			clearInterval(this.interval);
174 			clearInterval(this.interval);
175 			this.isRunning = false;
176 		},
177 
178 		/* ... */		
179 		reset : function(){
180 			$('#overlay').html('');
181 			this.annotations = [];
182 		},
183 	
184 		/* ... */
185 		parseTime : function (strTime) { 
186 			return strTime;
187 			var aTime = strTime.toString().split(":");
188 			return parseInt(aTime[0],10) * 60 + parseInt(aTime[1],10) * 1;// + parseFloat(aTime[2]);
189 		},
190 	
191 		/* ... */
192 		getCurrentTime : function(){
193 			return this.player.currentTime();
194 		}
195 	
196 }); // end class Clock
