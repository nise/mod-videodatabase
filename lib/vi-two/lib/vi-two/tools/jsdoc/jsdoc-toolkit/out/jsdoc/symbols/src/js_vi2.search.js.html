 10 var Vi_Search = $.inherit(/** @lends Search# */{
 11 
 12 	/** constructs */
 13 	__constructor : function() {// options

 15 			 
 16 	},
 17 	
 18 	name : 'search',
 19 	options : {selector: '#null'},
 20 	content_selector : '#content',
 21 	total : 0,
 22 
 23 
 24 	/* handle search */
 25 	find : function(string){
 26 		
 27 		if(ocr == undefined){ alert(22);
 28 			window.setTimeout("vi2.observer.widget_list[search].getStreamsBySearch('"+string+"')", 1000); // bug!!
 29 			return;
 30 		}else{
 31 			var t1 = 0, t2 = 0;
 32 			var t = new Date(); t1 = t.getTime();
 33 			var _this = this;
 34 			var result = [];
 35 			var maxResult = 1;
 36 		
 37 	/*		var t1 = 0; var t2 = 0;
 38 			$.ajax({type: "POST", dataType: "json", url: './ocr.json', 
 39 				beforeSend : function(){ 
 40 					var t = new Date(); t1 = t.getTime(); 
 41 				},
 42 				success: function(res){  
 43 					var t = new Date(); t2 = t.getTime();
 44 					_this.ocr = res;
 45 		*/			
 46 				// split search string into words ==> buggy
 47 				$.each(string.split(" "), function(i, str){ 
 48 						var expp = new RegExp(str, "gi");
 49 						// parse json completly 
 50 						$.each(vi2.db.json_data.stream, function(i, stream){ 
 51 								// add stream as potential result
 52 								stream.id = String(stream.id);
 53 								if(result[i] == null){ 
 54 									result[i] = {}; 
 55 								} 
 56 								result[i].id = stream.id;
 57 								result[i].title = stream.metadata[0].title;
 58 								result[i].author = stream.metadata[0].author;
 59 								if(result[i]['abstract'] == undefined){ result[i]['abstract'] = 0; }
 60 								if(result[i].tags == undefined){ result[i].tags = 0; }
 61 								if(result[i].auth == undefined){ result[i].auth = 0; }
 62 								if(result[i].fulltext == undefined){ result[i].fulltext = 0; }
 63 							
 64 								// increment title-count in case it matches
 65 								if(stream.metadata[0].title.search(expp) != -1 ){
 66 									result[i].ti = 1;
 67 								}else{
 68 									result[i].ti = 0;
 69 								}		
 70 								// increment author-count in case it matches
 71 								if(stream.metadata[0].author.search(expp) != -1 ){
 72 									result[i].auth = 1;
 73 								}else{
 74 									result[i].auth = 0;
 75 								}		
 76 								// increment abstract-count in case it matches
 77 								if(stream.metadata[0]['abstract'].search(expp) != -1 ){
 78 									result[i]['abstract'] += stream.metadata[0]['abstract'].match(expp).length;
 79 								}
 80 								// increment tag-count in case one or more are matching
 81 								var t = '';
 82 
 83 								$.each(stream.tags, function(i,val){ 
 84 									t += this.tagname+' '; 
 85 								});	
 86 					
 87 								if(t.search(expp) != -1 ){
 88 									result[i].tags += t.match(expp).length;
 89 									//alert(stream.id+' '+ result[stream.id].tags);
 90 								}
 91 							
 92 								// full text search inside slides 
 93 								//var slides = '';
 94 								$.each(ocr, function(index, val){ 
 95 									if(val.id == stream.id){ 
 96 										$.each(val.slides, function(index2, vall){ 
 97 											if((vall.text).search(expp) != -1){ 
 98 												result[i].fulltext++;
 99 												//slides = slides + ',' + vall.source;
100 											}
101 										});
102 									}	
103 								});
104 							
105 								// weight results 
106 								result[i]['total'] = result[i].ti * 12 + result[i].auth * 10 + result[i]['abstract'] * 2 + result[i].tags * 10 + result[i].fulltext * 2;
107 				
108 								if (result[i].total > maxResult) { 
109 									maxResult = result[i].total; 
110 								}
111 						});
112 					});
113 					// normalize result
114 					$.each(result, function(i, val){ if(this.total > 0) this.total = Math.ceil((100*this.total) / maxResult);});
115 					var t = new Date(); t2 = t.getTime(); 
116 					// render results, header and template
117 					$(_this.content_selector).trigger('clear').empty().append($('<h2></h2>').text('Search results for "'+string+'"')).append('<span class="des">(search time: '+(t2-t1)+' ms)</span>');
118 						var item =$('<div></div>')
119 							.setTemplate($("#search-template").val())
120 							.processTemplate({results:result}) // vi2.db.json_data.stream.stream
121 							.appendTo($(_this.content_selector));
122 
123 					// sort results
124 					$('div.search-results > div').tsort('.res', {order:"desc"});
125 
126 		
127 		}
128 		
129 	}

131 }); // end class
