 23 var Vi2_API = $.inherit(/** @lends API# */{
 24 	/**
 25 	* The Data API is implemented as an flexible interface between vi2.db.js and the address bar of the browser. More useful are api call via node.js. 
 26 	* Combinations of functions can be requested, each with multiple parameter. The output is presented as is. No further calculations or joins of the data are not part of the API. Sofar it is an output API only. Input to the database can not be handled. The default output format is JSON. Other formats are still under construction. 	
 27 	* 
 28 	* 	@constructs 
 29 	*/
 30   __constructor : function() { 
 31   	this.url = purl();
 32   },
 33   
 34   url : '',
 35   format : 'json',
 36   func_list : ['getStreamById', 'getCategoryTaxonomie'],
 37   
 38   /** ... */
 39   get : function(url){
 40 	  var _this = this;
 41   	// determine output format
 42   	this.format = this.url.fparam('format') != '' ? this.url.fparam('format') : this.format;
 43   	//
 44   	this.funcs = this.getRequestedFunctions();
 45   	
 46   	// cycle through all requested functions and collect results
 47   	$.each(this.funcs, function(i, func){ 
 48  			// get parameter
 49   		this.params = _this.url.fparam(func.name).split(',');
 50   		if(this.params.length == 0){
 51   			func.results['void'] = vi2.db[func.name]();
 52 			}else{
 53 				// cycle through param
 54 				$.each(this.params, function(j, param){ 
 55 					// call function and store results
 56 					func.results[param] = vi2.db[func.name](param); 
 57 				});
 58   		}
 59   		
 60   	});  	
 61   	this.render(this.funcs);
 62   },
 63   
 64   /** ... */
 65   getRequestedFunctions : function(){
 66   	var _this = this;
 67   	var functions = new Object();
 68   	$.each(this.func_list, function(i, func){
 69   		if(_this.url.fparam(func)){
 70   			functions[func] = {name: func, params: '', results: {}}; 
 71   		}
 72   	});
 73   	return functions;
 74   },
 75   
 76   /** 
 77   * @description Passes data with data URI scheme
 78   * @parameter {json} data
 79    */
 80   render : function(data){
 81   	switch(this.format){
 82   		case 'xml':
 83   			
 84   			break;
 85   		default:	// json
 86   			//return data;
 87   				window.location.href = 'data:application/json;charset=utf-8,' + encodeURIComponent( JSON.stringify(data) );
 88   			//$('body').empty().text(JSON.stringify(data));
 89   	}
 90   	
 91   }
 92   
 93   	
 94 
 95 	}); // end class API
