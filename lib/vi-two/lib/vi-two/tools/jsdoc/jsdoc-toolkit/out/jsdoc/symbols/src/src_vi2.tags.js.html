 91 		displayTagcloud : function(){  
 92 			$(this.options.selector).empty();
 93 			var _this = this;
 94 			// template for displaying tags
 95 			var ln = function(_name, _freq, _tags){ 
 96 				return $('<li></li>').addClass('list-item').attr('id', _name.replace(' ', '--')).append($('<a></a>').attr('freq', _freq).addClass('id-'+_name.replace(' ', '--'))

100 					.css("font-size", (_freq / 10 < 1) ? _freq / 10 + 1 + "em": (_freq / 10 > 2) ? "2em" : _freq / 10 + "em")
101 					.bind('click', {tags:_tags}, function(e){ 
102 						vi2.observer.log('clicktagfromlist:'+_name);
103 						_this.showTimelineTag(e.data);
104 					})
105 					);	
106 			};
107 			
108 			// prepare list and append existing tags
109 			var ul = $("<ul></ul>").addClass("tagslist").appendTo(this.options.selector);   
110 			$.each(this.tag_obj, function(i, val){
111 				ul.append(ln(val.tagname, val.occ.length, val));
112 			});
113 			// sort by occurence or alphabeticly, sort order desc / asc
114 			var sortAttr = {};
115 			sortAttr.attr = this.options.sort == 'freq' ? 'freq' : '';
116 			sortAttr.order = this.options.order == 'desc' ? "desc" : "asc";
117 			ul.find('a').tsort(sortAttr); 
118 			// cut off elements above max and render them
119 			$(ul).find('li:gt('+(this.options.max-1)+') > a').hide();
120

136 						.bind('click', function(){

138 						})
139 						;

143 				};

146 				$.each(e.tags.occ, function(){
147 					var progress = this / _this.player.duration();
148 					progress = ((progress) * $(_this.timelineSelector).width());
